2. 加密解密工具 (src/server/utils/crypto.ts)
typescript
import CryptoJS from 'crypto-js';

/**
 * 微信消息加解密工具类
 */
export class WechatCrypto {
  private token: string;
  private encodingAESKey: string;
  private key: CryptoJS.lib.WordArray;
  private iv: CryptoJS.lib.WordArray;

  constructor(token: string, encodingAESKey: string) {
    this.token = token;
    this.encodingAESKey = encodingAESKey;
    
    // 解码 Base64 的 EncodingAESKey
    const decodedKey = CryptoJS.enc.Base64.parse(encodingAESKey);
    this.key = decodedKey;
    
    // IV 是前 16 字节
    this.iv = CryptoJS.lib.WordArray.create(decodedKey.words.slice(0, 4), 16);
  }

  /**
   * 验证 GET 请求的签名（消息推送验证）
   */
  checkSignature(signature: string, timestamp: string, nonce: string): boolean {
    const arr = [this.token, timestamp, nonce].sort();
    const str = arr.join('');
    const hash = CryptoJS.SHA1(str).toString();
    return hash === signature;
  }

  /**
   * 解密 POST 请求的加密数据
   */
  decrypt(encryptedData: string): string {
    try {
      // Base64 解码
      const ciphertext = CryptoJS.enc.Base64.parse(encryptedData);
      
      // AES-256-CBC 解密
      const decrypted = CryptoJS.AES.decrypt(
        { ciphertext },
        this.key,
        {
          iv: this.iv,
          mode: CryptoJS.mode.CBC,
          padding: CryptoJS.pad.Pkcs7
        }
      );
      
      // 转换为 UTF-8 字符串
      const xmlText = CryptoJS.enc.Utf8.stringify(decrypted);
      
      // 移除微信添加的随机字节和长度信息
      const contentLength = parseInt(xmlText.substring(16, 20), 10);
      const content = xmlText.substring(20, 20 + contentLength);
      
      return content;
    } catch (error) {
      console.error('解密失败:', error);
      throw new Error('Failed to decrypt message');
    }
  }

  /**
   * 生成 POST 请求的 HMAC-SHA256 签名
   */
  generatePostSignature(timestamp: string, nonce: string, body: string): string {
    const message = timestamp + nonce + body;
    const hash = CryptoJS.HmacSHA256(message, this.token);
    return CryptoJS.enc.Hex.stringify(hash);
  }
}
3. Webhook 路由处理 (src/server/routes/webhook.ts)
typescript
import { Router, Request, Response } from 'express';
import { WechatCrypto } from '../utils/crypto';
import { channelsConfig } from '../config/channels';
import axios from 'axios';

const router = Router();
const wechatCrypto = new WechatCrypto(
  channelsConfig.token,
  channelsConfig.encodingAESKey
);

/**
 * 视频号小店 Webhook 入口
 */
router.all('/channels-webhook', async (req: Request, res: Response) => {
  const method = req.method.toUpperCase();
  
  try {
    if (method === 'GET') {
      // GET 请求：消息推送验证
      const { signature, timestamp, nonce, echostr } = req.query;
      
      if (!signature || !timestamp || !nonce || !echostr) {
        res.status(400).send('Missing required parameters');
        return;
      }
      
      const isValid = wechatCrypto.checkSignature(
        signature as string,
        timestamp as string,
        nonce as string
      );
      
      if (!isValid) {
        res.status(401).send('Invalid signature');
        return;
      }
      
      res.setHeader('Content-Type', 'text/plain');
      res.send(echostr);
      
    } else if (method === 'POST') {
      // POST 请求：接收订单事件
      const encryptedData = req.body;
      const timestamp = req.headers['x-wx-timestamp'] as string;
      const nonce = req.headers['x-wx-nonce'] as string;
      const signature = req.headers['x-wx-signature'] as string;
      
      // 验证 POST 签名
      const expectedSignature = wechatCrypto.generatePostSignature(
        timestamp,
        nonce,
        JSON.stringify(encryptedData)
      );
      
      if (signature !== expectedSignature) {
        res.status(401).send('Invalid signature');
        return;
      }
      
      // 解密数据并处理订单
      const decryptedData = JSON.parse(wechatCrypto.decrypt(encryptedData));
      const { Event, order_info } = decryptedData;
      
      if (Event === 'channels_ec_order_pay_success') {
        // 处理订单支付成功逻辑
        await handleOrderPaid(order_info);
      }
      
      res.send('success');
    }
  } catch (error) {
    console.error('Webhook 处理失败:', error);
    res.status(500).send('Internal Server Error');
  }
});

2.5 access_token 维护 + 虚拟商品发货接口 (src/server/services/channels-api.ts)

注意：发货接口调用走的是 access_token，而不是商户号+API密钥。
微信小店开放 API 的统一凭证是 access_token，通过 AppID/AppSecret 调用
stable_token 接口换取（与小程序支付场景的商户号体系不同）。

typescript
import { channelsConfig } from '../config';

const API_BASE = 'https://api.weixin.qq.com';

// access_token 内存缓存（进程级），提前 5 分钟刷新
let cachedAccessToken: { token: string; expiresAt: number } | null = null;
const TOKEN_REFRESH_LEAD_MS = 5 * 60 * 1000;

/**
 * 获取 access_token（带内存缓存，使用 stable_token 接口）
 *
 * stable_token 与普通 token 区别：
 * - 普通 token：每次获取都会使上一份失效，多端共用时互相挤掉
 * - stable_token：force_update=false 时不强制刷新，多端共存更稳
 *
 * @param forceUpdate 是否强制刷新（用于上游报 40001 token 失效时重试）
 */
export async function getChannelsAccessToken(forceUpdate = false): Promise<string> {
  // 缓存命中且未到期
  if (!forceUpdate && cachedAccessToken && Date.now() < cachedAccessToken.expiresAt) {
    return cachedAccessToken.token;
  }

  if (!channelsConfig.appId || !channelsConfig.appSecret) {
    throw new Error('[channels] 未配置 CHANNELS_APP_ID / CHANNELS_APP_SECRET');
  }

  const url = `${API_BASE}/cgi-bin/stable_token`;
  const resp = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      grant_type: 'client_credential',
      appid: channelsConfig.appId,
      secret: channelsConfig.appSecret,
      force_update: forceUpdate,
    }),
  });
  const data = await resp.json() as {
    access_token?: string;
    expires_in?: number;
    errcode?: number;
    errmsg?: string;
  };

  if (!data.access_token) {
    throw new Error(`[channels] 获取 access_token 失败: errcode=${data.errcode} errmsg=${data.errmsg}`);
  }

  cachedAccessToken = {
    token: data.access_token,
    expiresAt: Date.now() + (data.expires_in ?? 7200) * 1000 - TOKEN_REFRESH_LEAD_MS,
  };
  return cachedAccessToken.token;
}

/**
 * 虚拟商品发货（deliver_type=3，无需物流）
 *
 * 接口：POST /channels/ec/order/delivery_send
 * 文档：https://developers.weixin.qq.com/doc/channels/API/order/delivery_send.html
 *
 * 触发时机：订单支付成功后调用，标记订单为已发货
 *
 * @param orderId 微信小店订单号（order_id）
 */
export async function deliverVirtualOrder(orderId: string): Promise<boolean> {
  if (!channelsConfig.appId) {
    console.log('[channels] mock 跳过发货调用：orderId=', orderId);
    return false;
  }

  const token = await getChannelsAccessToken();
  const url = `${API_BASE}/channels/ec/order/delivery_send?access_token=${token}`;
  const resp = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      order_id: orderId,
      delivery_list: [
        {
          // 1=自寄快递，3=虚拟商品无需物流发货
          deliver_type: 3,
        },
      ],
    }),
  });
  const data = await resp.json() as { errcode?: number; errmsg?: string };
  if (data.errcode && data.errcode !== 0) {
    throw Object.assign(new Error(`发货失败: errcode=${data.errcode} errmsg=${data.errmsg}`), {
      errcode: data.errcode,
    });
  }
  return true;
}

/**
 * 发送客服文本消息（携带兑换码给买家）
 *
 * 接口：POST /cgi-bin/message/custom/send
 * 限制：用户在小店与客服有 48 小时交互窗口期内才能下发
 *
 * @param openid 买家 openid
 * @param content 文本内容（兑换码或跳转链接）
 */
export async function sendChannelsCustomMessage(openid: string, content: string): Promise<boolean> {
  if (!channelsConfig.appId || !openid) return false;

  const token = await getChannelsAccessToken();
  const url = `${API_BASE}/cgi-bin/message/custom/send?access_token=${token}`;
  const resp = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      touser: openid,
      msgtype: 'text',
      text: { content },
    }),
  });
  const data = await resp.json() as { errcode?: number; errmsg?: string };
  if (data.errcode && data.errcode !== 0) {
    console.warn(`[channels] 客服消息发送失败: errcode=${data.errcode} errmsg=${data.errmsg}`);
    return false;
  }
  return true;
}

5. 小程序解锁页面 (src/pages/video/unlock.tsx)
tsx
import { useEffect, useState } from 'react';
import { View, Button, Input, Text } from '@tarojs/components';
import Taro from '@tarojs/taro';
import axios from 'axios';

export default function UnlockPage() {
  const [code, setCode] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  
  useEffect(() => {
    const params = Taro.getCurrentInstance().router?.params;
    if (params?.code) {
      setCode(params.code);
      handleRedeem(params.code);
    }
  }, []);
  
  const handleRedeem = async (redeemCode?: string) => {
    const targetCode = redeemCode || code;
    
    if (!targetCode) {
      setError('请输入兑换码');
      return;
    }
    
    setLoading(true);
    setError('');
    
    try {
      const response = await axios.post('/api/redeem', { code: targetCode });
      
      if (response.data.success) {
        Taro.navigateTo({
          url: `/pages/video/play?collection_id=${response.data.collectionId}&token=${response.data.token}`
        });
      } else {
        setError(response.data.message || '兑换失败');
      }
    } catch (error) {
      setError('网络错误，请重试');
    } finally {
      setLoading(false);
    }
  };
  
  return (
    <View className='unlock-container'>
      <Text className='title'>兑换付费视频</Text>
      
      {!code && (
        <View className='input-section'>
          <Input
            value={code}
            onInput={(e) => setCode(e.detail.value)}
            placeholder='请输入兑换码'
            maxlength={20}
          />
          <Button onClick={() => handleRedeem()} loading={loading} disabled={!code}>
            立即兑换
          </Button>
        </View>
      )}
      
      {error && <Text className='error'>{error}</Text>}
    </View>
  );
}

常见问题排查

1. HTTP 请求连接失败
检查域名是否已备案
确认 443 端口是否对外开放
验证 SSL 证书是否有效
2. 验签失败
确认 Token 和 EncodingAESKey 是否正确
检查时间戳是否在有效期内（±300秒）
确保返回 echostr 是纯文本格式
3. 解密失败
确认 EncodingAESKey 是否为 43 位
检查解密算法是否为 AES-256-CBC
验证 IV 是否正确提取
4. 发货接口报错
确认商品类目是否为虚拟商品
检查 deliver_type是否为 3
验证 product_cnt是否与订单一致

5. 获取 access_token 失败 / 调用 API 报 40164
确认 CHANNELS_APP_ID / CHANNELS_APP_SECRET 是否填写正确（与小程序 AppID 不同）
检查服务器公网 IP 是否已加入微信小店后台「开发配置 → IP白名单」
如多端共用同一组凭证，建议用 stable_token + force_update=false，避免互相挤掉 token

6. access_token 失效（errcode 40001 / 42001）
本地缓存可能过期未及时刷新，调用方应捕获 40001/42001 后强制刷新一次重试
（项目实现见 services/channels-api.ts 的 callChannelsApiWithRetry）

7. 客服消息发送失败（errcode 10706 等）
用户在小店与客服的 48 小时交互窗口期已过，无法主动下发
建议同时把兑换码写入订单备注 / 短信通知，作为兜底通道
注意：本文档为技术实现指南，请根据实际业务需求调整代码逻辑，并确保遵守微信官方接口规范。
